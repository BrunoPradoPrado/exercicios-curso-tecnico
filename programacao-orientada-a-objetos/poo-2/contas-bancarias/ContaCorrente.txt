
public class ContaCorrente {
	private int numero;
	private String senha;
	private float saldo;

	
	public ContaCorrente(int numero, String senha) {
		super();
		this.numero = numero;
		this.alteraSenha(senha);
		this.saldo = 0;
	}

	public int getNumero() {
		return numero;
	}

	public void setNumero(int numero) {
		this.numero = numero;
	}

	public String getSenha() {
		return senha;
	}

	public void setSenha(String senha) {
		this.senha = senha;
	}

	public float getSaldo() {
		return saldo;
	}

	public void setSaldo(float saldo) {
		this.saldo = saldo;
	}

	public boolean sacar(float valor) {
		boolean r = false;
		if (this.getSaldo() >= valor) {
			this.saldo = this.saldo-valor;
			r = true;
		}
		return r;
	}
	public void depositar(float valor) {
		this.saldo = this.saldo+valor;
	}
	public void transferir(ContaCorrente c, float valor) {
		if(this.sacar(valor)) {
			c.depositar(valor);
		}
	}
	public void extrato() {
		System.out.println("Numero da conta: "+this.numero);
		System.out.println("Senha da conta: "+this.senha);
		System.out.println("Saldo da conta: "+this.saldo);
	}
	public void alteraSenha(String novaSenha) {
		if(this.validaSenha(novaSenha)) {
			this.senha=novaSenha;
		} else {
			System.out.println("Senha inválida. Senha '0000' atribuída.");
			this.senha = "0000";
		}
	}
	private boolean validaSenha(String novaSenha) {
		boolean r = false;
		if(novaSenha.length()==4) {
			r = true;
		}
		return r;
	}
}
